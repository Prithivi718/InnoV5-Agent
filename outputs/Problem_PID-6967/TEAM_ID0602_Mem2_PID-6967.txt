sensor_value = None
previous_value = None

def text_prompt(msg):
  try:
    return raw_input(msg)
  except NameError:
    return input(msg)


sensor_value = float(text_prompt(''))
previous_value = 0
if sensor_value == previous_value:
  print('Sensor update delayed, using previous value')
else:
  previous_value = sensor_value
  print('Sensor updated')

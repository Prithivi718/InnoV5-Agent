StabilityValue = None
SafetyValue = None
threshold_stability = None
threshold_safety = None

def text_prompt(msg):
  try:
    return raw_input(msg)
  except NameError:
    return input(msg)


StabilityValue = text_prompt('')
SafetyValue = text_prompt('')
if StabilityValue > threshold_stability and SafetyValue > threshold_safety:
    print('Stable and Safe')

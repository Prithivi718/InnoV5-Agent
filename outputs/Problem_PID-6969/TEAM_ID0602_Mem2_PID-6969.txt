counter = None
sensor_value = None

def text_prompt(msg):
  try:
    return raw_input(msg)
  except NameError:
    return input(msg)


counter = 0
for count in range(10):
  sensor_value = float(text_prompt(''))
  if sensor_value != 100:
    print('Sampling rate exceeded!')
  else:
    counter = counter + 1

command = None
min_limit = None
max_limit = None

def text_prompt(msg):
  try:
    return raw_input(msg)
  except NameError:
    return input(msg)


command = float(text_prompt(''))
min_limit = float(text_prompt(''))
max_limit = float(text_prompt(''))
if command >= min_limit and command <= max_limit:
    print('valid')

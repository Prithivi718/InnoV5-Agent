Role_A_Trust_Level = None
Role_B_Trust_Level = None

def text_prompt(msg):
  try:
    return raw_input(msg)
  except NameError:
    return input(msg)


Role_A_Trust_Level = float(text_prompt(''))
Role_B_Trust_Level = float(text_prompt(''))
if Role_A_Trust_Level > Role_B_Trust_Level:
    print('True')
